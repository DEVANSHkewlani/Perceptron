What Is the Chaos Lab
Purpose, components, and how everything fits together

The Chaos Lab is a controlled testing harness that sits between your DCA (Digital Cognitive Architecture) and a realistic target system (ShopCore). It has three jobs:

💥
Inject Real Failures
The Chaos Engine injects 18 types of real infrastructure failures — service crashes, slow DB queries, Kafka consumer lag, Redis eviction storms, network packet loss — using Docker env-var hooks and direct infrastructure manipulation.
📡
Trigger DCA Perception
Every injected failure produces authentic signals (structured logs → Fluent Bit → Kafka, Prometheus metrics → AlertManager, direct health-poll misses) that flow through the DCA's existing perception pipeline and produce CognitiveEvents.
🔧
Validate Autonomous Healing
New remediation handlers (service restart, DB kill slow query, cache flush, Kafka offset reset, config rollback) are registered with the DCA's action executor so the system can self-heal without human approval on critical scenarios.
📌
Scope of This Guide
This guide covers building and connecting the Chaos Engine only. ShopCore (the target system) is fully specified in the ShopCore Build Guide. The DCA perception, reasoning, and execution layers are built in Phases 1–11. Phase 13 wires them all together and adds the chaos injection + remediation layer.

ChaosController — chaos_engine/controller.py
Central orchestrator: activate, deactivate, verify, and track scenario state with timestamps

The ChaosController is the single object that owns all scenario state. It wraps two injection mechanisms: DockerChaos (env-var mutation + container restart) and NetworkChaos (tc/iptables). Every scenario registers activate, deactivate, and verify_injected callables here.

class ScenarioState(Enum):
    IDLE       = "idle"
    ACTIVE     = "active"
    RECOVERING = "recovering"
    FAILED     = "failed"


@dataclass
class ChaosScenario:
    name:                   str
    description:            str
    severity:               str     # critical | high | medium | low
    expected_event_type:    str     # maps to DCA CognitiveEvent.event_type
    expected_detection_s:   float   # detection SLO in seconds
    expected_remediation_s: float   # remediation SLO in seconds
    activate:               Callable[[dict], Awaitable[None]]
    deactivate:             Callable[[], Awaitable[None]]
    verify_injected:        Callable[[], Awaitable[bool]]
    state:        ScenarioState = ScenarioState.IDLE
    activated_at: datetime | None  = None
    detected_at:  datetime | None  = None
    remediated_at:datetime | None  = None
    params:       dict             = field(default_factory=dict)


class DockerChaos:
    """Inject failures by setting env vars on running containers."""

    def __init__(self):
        self.client = docker.from_env()

    def _get_container(self, service_name: str):
        # Try label first, fall back to name match
        containers = self.client.containers.list(
            filters={"label": f"com.shopcore.service={service_name}"}
        )
        if not containers:
            containers = self.client.containers.list(
                filters={"name": service_name}
            )
        if not containers:
            raise RuntimeError(f"Container not found: {service_name}")
        return containers[0]

    async def set_env(self, service: str, key: str, value: str) -> None:
        """Set an environment variable on a running container (Docker exec)."""
        container = self._get_container(service)
        container.exec_run(["sh", "-c", f"export {key}={value}"], detach=True)
        # Some services read env at request time — set via docker update
        await asyncio.to_thread(
            container.client.api.update_container,
            container.id,
            Env=[f"{key}={value}"]
        )
        logger.info(f"[chaos] {service}: {key}={value}")

    async def restart(self, service: str) -> None:
        container = self._get_container(service)
        await asyncio.to_thread(container.restart, timeout=5)
        logger.info(f"[chaos] restarted {service}")

    async def kill(self, service: str) -> None:
        container = self._get_container(service)
        await asyncio.to_thread(container.kill)
        logger.info(f"[chaos] killed {service}")

    async def is_running(self, service: str) -> bool:
        try:
            c = self._get_container(service)
            return c.status == "running"
        except RuntimeError:
            return False


class ChaosController:
    """
    Top-level orchestrator. Owns all scenario state.
    Called by the REST API (api.py) and the Scheduler (scheduler.py).
    """

    def __init__(self, scenarios: dict[str, ChaosScenario]):
        self.scenarios = scenarios
        self.docker    = DockerChaos()
        self._active: set[str] = set()

    async def activate(self, name: str, params: dict | None = None) -> None:
        s = self.scenarios[name]
        if s.state == ScenarioState.ACTIVE:
            logger.warning(f"[chaos] {name} already active — skipping")
            return
        logger.info(f"[chaos] activating {name}")
        s.params = params or {}
        await s.activate(s.params)
        s.state = ScenarioState.ACTIVE
        s.activated_at = datetime.now(timezone.utc)
        s.detected_at  = None
        s.remediated_at = None
        self._active.add(name)
        logger.info(f"[chaos] {name} active — waiting for DCA detection")

    async def deactivate(self, name: str) -> None:
        s = self.scenarios[name]
        await s.deactivate()
        s.state = ScenarioState.IDLE
        self._active.discard(name)
        logger.info(f"[chaos] {name} deactivated")

    async def mark_detected(self, name: str) -> None:
        s = self.scenarios[name]
        s.detected_at = datetime.now(timezone.utc)
        if s.activated_at:
            dt = (s.detected_at - s.activated_at).total_seconds()
            logger.info(f"[chaos] {name} detected in {dt:.1f}s (SLO: {s.expected_detection_s}s)")

    async def mark_remediated(self, name: str) -> None:
        s = self.scenarios[name]
        s.remediated_at = datetime.now(timezone.utc)
        s.state = ScenarioState.RECOVERING
        if s.activated_at:
            dt = (s.remediated_at - s.activated_at).total_seconds()
            logger.info(f"[chaos] {name} remediated in {dt:.1f}s (SLO: {s.expected_remediation_s}s)")

    def get_status(self) -> list[dict]:
        return [
            {
                "name":            s.name,
                "state":           s.state.value,
                "severity":        s.severity,
                "expected_event":  s.expected_event_type,
                "activated_at":    s.activated_at.isoformat() if s.activated_at else None,
                "detected_at":     s.detected_at.isoformat()  if s.detected_at  else None,
                "remediated_at":   s.remediated_at.isoformat() if s.remediated_at else None,
            }
            for s in self.scenarios.values()
        ]

async def _check_service_down(url: str) -> bool:
    try:
        async with httpx.AsyncClient(timeout=3) as c:
            r = await c.get(url)
            return r.status_code != 200
    except:
        return True

async def _check_latency_high(url: str, threshold_s: float) -> bool:
    import time
    start = time.monotonic()
    try:
        async with httpx.AsyncClient(timeout=10) as c:
            await c.get(url)
        return time.monotonic() - start >= threshold_s
    except:
        return True

async def _cascade_activate(d: DockerChaos):
    await asyncio.gather(
        d.kill("postgres"),
        d.set_env("order-service", "CHAOS_KAFKA_BLOCK", "true"),
        d.set_env("api-gateway", "CHAOS_ERROR_RATE", "0.9"),
    )

async def _cascade_deactivate(d: DockerChaos):
    await asyncio.gather(
        d.restart("postgres"),
        d.set_env("order-service", "CHAOS_KAFKA_BLOCK", "false"),
        d.set_env("api-gateway", "CHAOS_ERROR_RATE", "0"),
    )

REMEDIATION_TIME = Histogram("chaos_remediation_seconds", "Time from activate to full remediation", ["scenario"])

@asynccontextmanager
async def lifespan(app: FastAPI):
    global controller
    docker_chaos = DockerChaos()
    db_chaos     = DatabaseChaos(dsn="postgresql://shopcore:shopcore@localhost:5432/shopcore")

class ActivateRequest(BaseModel):
    params: dict = {}


@app.get("/scenarios")
async def list_scenarios():

    ACTIVATIONS.labels(scenario=name).inc()

@app.post("/scenarios/{name}/deactivate")

        if s.detected_at:

async def health(): return {"status": "ok", "service": "chaos-engine"}

import asyncio, random, logging, argparse, httpx, time
from datetime import datetime, timezone

logger = logging.getLogger("chaos-scheduler")

ALL_SCENARIOS = [
    "product_service_crash", "slow_database_query", "db_pool_exhaustion",
    "gateway_high_latency", "gateway_high_error_rate", "circuit_breaker_open",
    "order_service_slow", "kafka_producer_blocked", "order_error_spike",
    "auth_service_failure", "cache_miss_storm", "redis_memory_full",
    "redis_eviction_storm", "kafka_consumer_lag", "database_lock_contention",
    "notification_consumer_slow", "gateway_oom",
    # cascade_failure excluded from random gauntlet — run manually
]

CHAOS_API = "http://localhost:9090"


async def run_gauntlet(
    mode: str = "random",
    interval_min: int = 120,
    interval_max: int = 300,
    duration_minutes: int = 60,
    max_concurrent: int = 1,
    inject_duration_s: int = 120,
) -> None:
    deadline = time.monotonic() + duration_minutes * 60
    scenario_list = ALL_SCENARIOS[:]
    if mode == "sequential":
        random.shuffle(scenario_list)
    iteration = 0
    active: list[str] = []

    async with httpx.AsyncClient() as client:
        while time.monotonic() < deadline:
            iteration += 1

            # Pick next scenario
            if mode == "sequential":
                name = scenario_list[(iteration - 1) % len(scenario_list)]
            else:
                name = random.choice(scenario_list)

            logger.info(f"[gauntlet] iter={iteration} → activating {name}")
            await client.post(f"{CHAOS_API}/scenarios/{name}/activate")
            active.append(name)

            # Wait for DCA to detect + remediate
            await asyncio.sleep(inject_duration_s)

            # Deactivate (safety cleanup even if DCA already healed)
            await client.post(f"{CHAOS_API}/scenarios/{name}/deactivate")
            active.remove(name)

            # Cool-down before next injection
            wait = random.randint(interval_min, interval_max)
            logger.info(f"[gauntlet] cool-down {wait}s before next scenario")
            await asyncio.sleep(wait)

import asyncio, logging, docker

logger = logging.getLogger("network-chaos")

class NetworkChaos:
    """
    Uses Linux tc (traffic control) inside containers for real packet-level chaos.
    Container must run in privileged mode or have NET_ADMIN capability.
    """

    def __init__(self):
        self.client = docker.from_env()

    async def add_latency(self, service: str, latency_ms: int, jitter_ms: int = 10) -> None:
        """Inject network latency via tc netem inside the container."""
        cmd = (f"tc qdisc add dev eth0 root netem "
               f"delay {latency_ms}ms {jitter_ms}ms distribution normal")
        await self._exec(service, cmd)
        logger.info(f"[network-chaos] {service}: latency {latency_ms}ms ±{jitter_ms}ms")

    async def add_packet_loss(self, service: str, loss_pct: int) -> None:
        cmd = f"tc qdisc add dev eth0 root netem loss {loss_pct}%"
        await self._exec(service, cmd)

    async def remove(self, service: str) -> None:
        """Remove all tc rules — restore clean networking."""
        await self._exec(service, "tc qdisc del dev eth0 root 2>/dev/null || true")
        logger.info(f"[network-chaos] {service}: network rules removed")

    async def _exec(self, service: str, cmd: str) -> None:
        containers = self.client.containers.list(filters={"name": service})
        if not containers:
            raise RuntimeError(f"Container not found: {service}")
        container = containers[0]
        await asyncio.to_thread(container.exec_run, ["sh", "-c", cmd], privileged=True)

        """Hold an exclusive lock to simulate lock wait timeouts."""
        conn = await asyncpg.connect(self.dsn)
        try:
            await conn.execute(f"BEGIN; LOCK TABLE {table} IN EXCLUSIVE MODE;")
            logger.info(f"[db-chaos] lock held on {table} — will release after 120s")
            await asyncio.sleep(120)
        finally:
            await conn.execute("ROLLBACK")
            await conn.close()

    async def check_pool_exhausted(self, db_name: str) -> bool:
        """Return True if active connections ≥ max_connections - 2."""
        conn = await asyncpg.connect(self.dsn)
        row = await conn.fetchrow(
            """SELECT count(*) AS active,
                      current_setting('max_connections')::int AS max_c
               FROM   pg_stat_activity
               WHERE  datname = $1""",
            db_name
        )
        await conn.close()
        return row["active"] >= row["max_c"] - 2

class ServiceHandler:
    def __init__(self):
        self.client = docker.from_env()

    async def restart_service(self, ctx: dict) -> dict:
        """
        ctx must contain: {"service_id": "product-service"}
        Restarts the named Docker container and waits for health check.
        """
        service = ctx.get("service_id") or ctx.get("entity_id", "").replace("svc:", "")
        containers = self.client.containers.list(filters={"name": service})
        if not containers:
            logger.error(f"[service-handler] container not found: {service}")
            return {"status": "error", "reason": "container_not_found"}

        logger.info(f"[service-handler] restarting {service}")

        health_url = f"http://localhost:{ctx.get('port', 8001)}/health"

            except:
                pass

        logger.error(f"[service-handler] {service} did not recover after restart")
        return {"status": "timeout", "service": service}

        """Scale up replicas via docker compose. For now, restart is the proxy."""

import asyncpg, logging

logger = logging.getLogger("database-handler")

class DatabaseHandler:
    def __init__(self, dsn: str = "postgresql://shopcore:shopcore@localhost:5432/shopcore"):
        self.dsn = dsn

    async def kill_blocking_query(self, ctx: dict) -> dict:
        """
        Kill queries running longer than the threshold (default 5s).
        Returns count of queries killed.
        """
        threshold_s = ctx.get("threshold_s", 5)
        conn = await asyncpg.connect(self.dsn)
        try:
            rows = await conn.fetch(
                """
                SELECT pg_terminate_backend(pid) AS killed, pid, query
                FROM   pg_stat_activity
                WHERE  state = 'active'
                  AND  query_start < now() - ($1 * interval '1 second')
                  AND  pid <> pg_backend_pid()
                """,
                threshold_s
            )
            killed = [r for r in rows if r["killed"]]
            logger.info(f"[db-handler] killed {len(killed)} blocking queries")
            return {"status": "ok", "killed": len(killed)}
        finally:
            await conn.close()

    async def reset_pool(self, ctx: dict) -> dict:
        """Terminate all idle connections to force pool reset."""
        conn = await asyncpg.connect(self.dsn)
        try:
            rows = await conn.fetch(
                """
                SELECT pg_terminate_backend(pid)
                FROM   pg_stat_activity
                WHERE  state = 'idle'
                  AND  pid <> pg_backend_pid()
                  AND  datname = current_database()
                """
            )
            logger.info(f"[db-handler] reset pool — terminated {len(rows)} idle connections")
            return {"status": "ok", "terminated": len(rows)}
        finally:
            await conn.close()

    async def set_statement_timeout(self, ctx: dict) -> dict:
        """Apply a statement_timeout to stop runaway queries."""
        timeout_ms = ctx.get("timeout_ms", 2000)
        conn = await asyncpg.connect(self.dsn)
        try:
            await conn.execute(f"ALTER ROLE shopcore SET statement_timeout = '{timeout_ms}ms'")
            logger.info(f"[db-handler] statement_timeout set to {timeout_ms}ms")
            return {"status": "ok", "timeout_ms": timeout_ms}
        finally:
            await conn.close()

# Canonical "clean" env vars for each service — what we roll back TO
SAFE_ENV: dict[str, dict[str, str]] = {
    "api-gateway":     {"CHAOS_LATENCY_MS": "0", "CHAOS_ERROR_RATE": "0", "CHAOS_CIRCUIT_OPEN": "false"},
    "product-service": {"CHAOS_SLOW_QUERY_MS": "0", "CHAOS_POOL_EXHAUST": "false", "CHAOS_DB_ERROR_RATE": "0"},
    "order-service":   {"CHAOS_ORDER_DELAY_MS": "0", "CHAOS_KAFKA_BLOCK": "false", "CHAOS_ORDER_FAIL_RATE": "0"},
    "user-service":    {"CHAOS_AUTH_FAIL_RATE": "0"},
    "cart-service":    {"CHAOS_CACHE_MISS": "false", "CHAOS_CART_FAIL_RATE": "0"},
}

class ConfigHandler:
    def __init__(self):
        self.docker = docker.from_env()

    async def rollback_config(self, ctx: dict) -> dict:
        """Reset all chaos env vars for a service to their safe defaults."""
        service = ctx.get("service_id", "").replace("svc:", "")
        safe = SAFE_ENV.get(service, {})
        if not safe:
            logger.warning(f"[config-handler] no safe env map for {service}")
            return {"status": "skipped"}

        containers = self.docker.containers.list(filters={"name": service})
        if not containers:
            return {"status": "error", "reason": "not_found"}

        container = containers[0]
        env_list = [f"{k}={v}" for k, v in safe.items()]
        await asyncio.to_thread(container.client.api.update_container, container.id, Env=env_list)
        logger.info(f"[config-handler] rolled back {service} to safe env")
        return {"status": "ok", "service": service, "reset_keys": list(safe.keys())}

    async def reload_config(self, ctx: dict) -> dict:
        """Rollback config then restart the service."""
        await self.rollback_config(ctx)
        containers = self.docker.containers.list(filters={"name": ctx.get("service_id", "")})
        if containers:
            await asyncio.to_thread(containers[0].restart, timeout=30)
        return {"status": "ok", "action": "reload_config"}

How Chaos Engine Reaches ShopCore

The Chaos Engine runs inside the same Docker network (shopcore) as all ShopCore services. It connects three ways:

1️⃣
Docker API — Container Env Mutation
The Chaos Engine mounts /var/run/docker.sock so it can call docker update on running containers to mutate env vars like CHAOS_SLOW_QUERY_MS=3000. ShopCore services read these on every request (not at startup), so the failure is live-injected with zero restart.
2️⃣
Direct asyncpg — Database Injection
For DB scenarios (lock contention, pool exhaustion), the Chaos Engine opens its own asyncpg connection to postgresql://shopcore:shopcore@postgres:5432/shopcore and runs SQL directly: LOCK TABLE products, pg_terminate_backend, ALTER ROLE SET statement_timeout.
3️⃣
AIOKafkaAdminClient — Queue Injection
For Kafka scenarios, the Chaos Engine uses AIOKafkaAdminClient connecting to redpanda:9092 to delete consumer group offsets (triggering lag) or stop consumer groups externally.

"""Run after the test suite to print a Phase 13 pass/fail scorecard."""
import json, sys
from pathlib import Path

def score(results_json: str):
    results      = json.loads(Path(results_json).read_text())
    det_pass     = results["detection_pass"]
    det_total    = results["detection_total"]
    remed_pass   = results["remediation_pass"]
    remed_total  = results["remediation_total"]

    pct_d = det_pass   / det_total   * 100
    pct_r = remed_pass / remed_total * 100

    print(f"\n{'═'*52}")
    print(f"  PHASE 13 CHAOS LAB SCORECARD")
    print(f"{'═'*52}")
    print(f"  Detection:    {det_pass}/{det_total} ({pct_d:.0f}%)  {'✅ PASS' if pct_d >= 90 else '❌ FAIL'}")
    print(f"  Remediation:  {remed_pass}/{remed_total} ({pct_r:.0f}%)  {'✅ PASS' if pct_r >= 75 else '❌ FAIL'}")
    phase_pass = pct_d >= 90 and pct_r >= 75
    print(f"\n  PHASE 13: {'✅ PASSED — Ready for Phase 14' if phase_pass else '❌ FAILED — Fix SLOs above'}")
    print(f"{'═'*52}\n")
    return phase_pass

if __name__ == "__main__":
    sys.exit(0 if score(sys.argv[1]) else 1)
