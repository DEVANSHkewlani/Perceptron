Phase 02

Unit tests

Test every normalizer as a pure function — raw dict in, assert the output. Zero infrastructure. This catches 80% of bugs before you ever touch Kafka.

No Docker needed · runs in ~2 seconds
1
Create test file
bash
mkdir tests && touch tests/__init__.py tests/test_normalizers.py
2
Paste test suite into test_normalizers.py
python · tests/test_normalizers.py
import pytest
from cognitive_perception.normalizers.log_normalizer import LogNormalizer
from cognitive_perception.normalizers.api_normalizer import APINormalizer
from cognitive_perception.normalizers.sensor_normalizer import SensorNormalizer
from cognitive_perception.normalizers.security_normalizer import SecurityEventNormalizer
from cognitive_perception.normalizers.metric_normalizer import MetricNormalizer
from cognitive_perception.schema.event import CognitiveEvent, PerceptionFailure

def test_log_db_timeout():
    result = LogNormalizer().normalize({
        "message": "connection to postgres-primary timed out after 5000ms",
        "level": "error", "service": "auth-service",
        "timestamp": "2024-01-15T14:23:11Z"
    }, "svc:auth-service")
    assert isinstance(result, CognitiveEvent)
    assert result.event_type == "database_connection_timeout"
    assert result.severity == "high"

def test_log_bad_input_returns_failure():
    result = LogNormalizer().normalize({"level": "info"}, "svc:auth-service")
    assert isinstance(result, PerceptionFailure)
    assert result.normalizer == "LogNormalizer"

def test_api_ssl_expiring():
    result = APINormalizer().normalize({
        "url": "https://example.com", "status_code": 200,
        "latency_ms": 100, "ok": True, "error": None,
        "ssl_days_remaining": 5, "latency_warn_ms": 500, "latency_high_ms": 2000
    }, "svc:example")
    assert result.event_type == "service_ssl_expiring"
    assert result.severity == "high"

def test_sensor_temperature_critical():
    result = SensorNormalizer().normalize({
        "sensor_type": "temperature", "value": 50.0,
        "location": "server-room",
        "thresholds": {"warn": 35.0, "critical": 45.0}
    }, "sensor:server-room-temp")
    assert result.event_type == "temperature_critical"
    assert result.severity == "critical"

def test_security_sql_injection():
    result = SecurityEventNormalizer().normalize({
        "type": "SQL_INJECTION", "ip": "1.2.3.4",
        "uri": "/api/users?id=1 OR 1=1", "country": "RU"
    }, "security:cloudflare")
    assert result.event_type == "sql_injection_attempt"
    assert "security:ip-1-2-3-4" in result.entity_refs

def test_metric_cpu_spike():
    result = MetricNormalizer().normalize({
        "labels": {"alertname": "HighCPUUsage", "instance": "node-01:9100", "job": "node"},
        "annotations": {"summary": "CPU above 85%"},
        "status": "firing", "startsAt": "2024-01-15T14:00:00Z"
    }, "metric:node-01")
    assert result.event_type == "cpu_spike"
    assert result.confidence == 0.98
3
Run the tests
bash
pytest tests/test_normalizers.py -v
✓All 6 tests pass in under 2 seconds. If an import fails, check that all __init__.py files exist.
Phase 03

Normalizer REPL

Run any normalizer interactively in a Python shell. Faster than writing a test every time you want to check one thing.

No Docker needed · instant
1
Open Python shell and try the log normalizer
python shell
from cognitive_perception.normalizers.log_normalizer import LogNormalizer
n = LogNormalizer()
r = n.normalize({
    "message": "connection pool exhausted — all 100 connections in use",
    "level": "error", "service": "order-service"
}, "svc:order-service")
print(r.event_type)    # database_connection_timeout
print(r.severity)     # high
print(r.entity_refs)  # ['svc:order-service', 'db:...']
print(r.confidence)   # 0.82
print(r.model_dump_json(indent=2))  # full JSON output
2
Test that bad input never crashes — returns PerceptionFailure
python shell
from cognitive_perception.normalizers.metric_normalizer import MetricNormalizer
r = MetricNormalizer().normalize({"labels": {}, "status": "firing"}, "metric:node-01")
print(type(r))            # <class 'PerceptionFailure'>
print(r.error_message)    # Alertmanager payload missing 'alertname' label
print(r.normalizer)       # MetricNormalizer
3
Inspect security event entity_refs
python shell
from cognitive_perception.normalizers.security_normalizer import SecurityEventNormalizer
r = SecurityEventNormalizer().normalize({
    "type": "BRUTE_FORCE", "ip": "192.168.1.100",
    "request_count": 1500, "country": "CN"
}, "security:fail2ban")
print(r.entity_refs)
# ['security:fail2ban', 'security:ip-192-168-1-100', 'security:country-cn']
print(r.severity)   # high (auto-elevated because request_count > 1000)

Publish test raw logs (simulates Fluent Bit)
python · terminal 2
import asyncio, json
from aiokafka import AIOKafkaProducer

TEST_LOGS = [
    {"tag":"auth-service","level":"error",
     "message":"connection to postgres-primary timed out after 5000ms",
     "timestamp":"2024-01-15T14:23:11Z","trace_id":"trace-001"},
    {"tag":"auth-service","level":"warn",
     "message":"JWT token is expired, rejecting request",
     "timestamp":"2024-01-15T14:23:15Z"},
    {"tag":"order-service","level":"fatal",
     "message":"worker process crashed unexpectedly",
     "timestamp":"2024-01-15T14:23:20Z"},
    {"tag":"auth-service","level":"error"},  # no message → PerceptionFailure
]

async def publish():
    p = AIOKafkaProducer(bootstrap_servers="localhost:9092")
    await p.start()
    for log in TEST_LOGS:
        await p.send_and_wait("raw.logs", json.dumps(log).encode())
    await p.stop()
    print("Published 4 test logs to raw.logs")

asyncio.run(publish())
4
Run the LogAdapter (terminal 3)
python · terminal 3
import asyncio
from cognitive_perception.adapters.log_adapter import LogAdapter, LogSourceConfig

sources = [
    LogSourceConfig(fluent_bit_tag="auth-service",  source_id="svc:auth-service"),
    LogSourceConfig(fluent_bit_tag="order-service", source_id="svc:order-service"),
]
asyncio.run(LogAdapter(sources).run())
ℹSwitch to the Redpanda Console browser tab now. You'll see 3 events appear on cognitive.events and 1 PerceptionFailure on cognitive.perception_failures.

python · terminal 4
import asyncio, json
from aiokafka import AIOKafkaConsumer

async def read():
    c = AIOKafkaConsumer("cognitive.events", bootstrap_servers="localhost:9092",
        auto_offset_reset="earliest", consumer_timeout_ms=5000, group_id="verify")
    await c.start()
    async for msg in c:
        e = json.loads(msg.value)
        print(f"  {e['event_type']:40} {e['severity']:8} conf:{e['confidence']}")
    await c.stop()

asyncio.run(read())

Console reference

What each console section tells you

The Redpanda Console has four views. Each one answers a different question about your pipeline's health.

● Topics → cognitive.events
Your main output. Every CognitiveEvent from every adapter appears here. Click any row to see the full JSON — event_type, severity, entity_refs, confidence, tags. This is how you verify "did my adapter produce the right event?"
● Topics → perception_failures
Every PerceptionFailure lands here. Contains the normalizer name, exact error message, and the raw_input that caused it. High count on one normalizer = schema changed upstream or a bug in that normalizer.
● Topics → raw.logs
Raw unprocessed logs from Fluent Bit (or your test scripts). Compare this with cognitive.events to see exactly what the LogNormalizer extracted from each message.
● Consumer groups
Shows lag per consumer group. Lag near 0 = adapter is keeping up. Growing lag = adapter is processing slower than producers are writing. First sign of a performance problem.
